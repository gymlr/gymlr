## <p align="center">Usage</p>

```ts
import { getHealth } from "gymlr";

function printHealth() {
    const health = getHealth();
    console.log(health);
}

printHealth();
```

```ts
import { getBoost } from "gymlr"

function printBoost() {
    const boost = getBoost();
    console.log(boost);
}

printBoost();
```
