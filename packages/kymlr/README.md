## <p align="center">Usage</p>

```kt
fun main() {
    val health = getHealth()

    health.forEach {
        println(it)
    }
}
```

```kt
fun main() {
    val boost = getBoost()

    boost.forEach {
        println(it)
    }
}
```

```kt
fun main() {
    val muscle = getMuscle()

    muscle.forEach {
        println(it)
    }
}
```
